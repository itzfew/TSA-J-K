function loadQuestionDetail() { const urlParams = new URLSearchParams(window.location.search); const [subject, chapter, questionId] = urlParams.get('id').split('/'); fetch(chaptersUrl) .then(response => response.json()) .then(data => { let question = {}; if (subject === 'biology') { question = data[0].questions.find(q => q.question_id === questionId); } else if (subject === 'physics') { question = data[1].questions.find(q => q.question_id === questionId); } else if (subject === 'chemistry') { question = data[2].questions.find(q => q.question_id === questionId); } if (question) { let questionDetailHTML = `

${question.content}

    ${question.options.map((opt, index) => `
  • ${String.fromCharCode(65 + index)}. ${opt.content}
    • ` ).join('')}
`; // Render any inline math (MathJax will handle it) questionDetailHTML = questionDetailHTML.replace(/\((.*?)\)/g, function(match, p1) { return `\\(${p1}\\)`; // Wrap the text inside \( \) for MathJax inline }); document.getElementById('question-detail').innerHTML = questionDetailHTML; // Ensure MathJax processes the inline math MathJax.Hub.Queue(["Typeset", MathJax.Hub]); // Add event listeners for options const options = document.querySelectorAll('.options li'); options.forEach(option => { option.addEventListener('click', function () { options.forEach(opt => opt.classList.remove('selected')); option.classList.add('selected'); disableOptions(options); }); }); // Start the timer startTimer(); } else { document.getElementById('question-detail').innerHTML = '

Question not found.

'; } }) .catch(error => { console.error('Error loading question details:', error); document.getElementById('question-detail').innerHTML = '

Error loading question details. Please try again later.

'; }); }
